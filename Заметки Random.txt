async Task GetGooglePageAsync() {

Главный поток вызывает таску в строке 1, с этого момента она работает параллельно в другом потоке. Главный поток идёт дальше до 11й строчки и встаёт, дожидаясь окончания работы таски. Сама таска в своём потоке идёт до 24й строки и встаёт, дожидаясь окончания запроса, после чего метод завершается в 26й, второй поток умирает, главный дожидается его окончания и идёт на 13ю.

void vga_load_font(const uint8_t *font_data) {
    // Отключаем защиту знакогенератора
    outb(0x3C4, 0x04);
    outb(0x3C5, 0x00);

    // Загружаем 256 символов × 16 байт
    for (int i = 0; i < 256 * 16; i++) {
        outb(0x3C6, font_data[i]);
    }

    // Включаем новую таблицу
    outb(0x3C4, 0x03);
    outb(0x3C5, 0x04);
}
